##Comando SQL para Buscar Clientes em SÃ£o Paulo

SELECT
    c.id_cliente,
    c.razao_social,
    t.numero AS telefone
FROM
    Clientes c
JOIN
    Estados e ON c.id_estado = e.id_estado
JOIN
    Telefones t ON c.id_cliente = t.id_cliente
WHERE
    e.sigla = 'SP';
